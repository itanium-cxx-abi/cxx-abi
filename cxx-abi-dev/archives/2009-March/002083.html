Are you talking about something simple like this use of sizeof being
incompatible?

void foo<int>(int*, A<sizeof (int)>) => _Z3fooIiEvPT_1AIXstS0_EE

template <unsigned long size> struct A {};
template <class T>
void foo(T*, A<sizeof(T)>) {}
int main() {
   A<sizeof(int)> x;
   foo<int>(0, x);
}
